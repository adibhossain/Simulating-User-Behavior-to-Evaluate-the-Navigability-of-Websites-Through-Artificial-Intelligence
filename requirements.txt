# Install:  pip install -r requirements.txt

# Standard-library modules used by the code need NO install:
#   time, math, random, io, subprocess, pickle

selenium                # drives Firefox, finds links/buttons, screenshots

numpy                   # math for curved mouse paths
sentence-transformers   # semantic text matching -> pulls in torch, transformers
networkx                # builds the navigation graph
matplotlib              # renders the graph image
reportlab               # generates report.pdf
